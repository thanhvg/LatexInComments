#+begin_src elisp

(defun find-comment-regions (beg end &optional merge-contiguous)
  "Find all comment regions between BEG and END.
Returns a list of elements (START END).
If MERGE-CONTIGUOUS is non-nil (or when called interactively),
consecutive comments separated only by whitespace/newlines are
merged into a single region."
  (interactive "r\np")
  (let ((regions nil)
        (pos beg))
    (save-excursion
      (goto-char beg)
      (while (< (point) end)
        (let ((state (syntax-ppss)))
          (cond
           ;; 1. If currently inside a comment, find its end
           ((nth 4 state)
            (let ((start (nth 8 state)))
              (parse-partial-sexp (point) end nil nil state 'comment)
              (push (list start (point)) regions)
              ;; Guarantee forward progress
              (when (= pos (point)) (forward-char 1))))
           ;; 2. Not in a comment: scan forward to the next comment start
           (t
            (let ((next-state (parse-partial-sexp (point) end nil nil state 'comment)))
              (if (nth 4 next-state)
                  (goto-char (nth 8 next-state))
                (goto-char end))))))
        (setq pos (point))))
    
    (setq regions (nreverse regions))
    
    ;; Merge adjacent comments separated only by whitespace/newlines
    (if (and regions merge-contiguous)
        (let ((merged (list (car regions))))
          (dolist (r (cdr regions))
            (let ((prev (car merged)))
              ;; Check if the gap between comments is only whitespace
              (if (save-excursion
                    (goto-char (cadr prev))
                    (skip-chars-forward " \t\r\n" (car r))
                    (= (point) (car r)))
                  (setcar merged (list (car prev) (cadr r)))
                (push r merged))))
          (nreverse merged))
      regions)))
#+end_src

1. *=syntax-ppss=*: Determines if the current point is inside a comment (=nth 4= of the syntax state is non-nil), and if so, finds the exact starting position of the comment (=nth 8=).
2. *=parse-partial-sexp=*: Uses the =comment= stop-argument to scan forward directly to the end of the current comment, or jump directly to the start of the next one.
3. *Merging*: If =merge-contiguous= is enabled, it automatically combines separate single-line comments (like multiple =;;= lines) into one continuous block if there is nothing but whitespace between them.

(defun find-comment-regions-treesit (beg end &optional merge-contiguous)
  "Find continuous comment regions between BEG and END using Tree-sitter.
Returns a list of (START END) pairs.
If MERGE-CONTIGUOUS is non-nil, consecutive comments separated
only by whitespace or newlines are merged into single regions."
  (interactive "r\np")
  (unless (and (fboundp 'treesit-available-p) (treesit-available-p))
    (user-error "Tree-sitter is not available in this Emacs session"))
  (let ((root (treesit-buffer-root-node)))
    (unless root
      (user-error "No Tree-sitter parser is active in this buffer"))
    (let (comments)
      ;; 1. Traverse AST to collect comment nodes overlapping with [beg, end]
      (cl-labels ((walk (node)
                    (let ((n-start (treesit-node-start node))
                          (n-end (treesit-node-end node)))
                      (when (and (<= n-start end) (>= n-end beg))
                        (if (string-match-p "comment" (treesit-node-type node))
                            (push (list (max n-start beg) (min n-end end)) comments)
                          ;; Recurse into children
                          (dolist (child (treesit-node-children node))
                            (walk child)))))))
        (walk root))
      
      (setq comments (nreverse comments)) ; Restore original document order
      
      ;; 2. Optionally merge contiguous comment lines (e.g., adjacent `;;` or `#`)
      (if (and comments merge-contiguous)
          (let ((merged (list (car comments))))
            (dolist (r (cdr comments))
              (let ((prev (car merged)))
                (if (save-excursion
                      (goto-char (cadr prev))
                      (skip-chars-forward " \t\r\n" (car r))
                      (= (point) (car r)))
                    (setcar merged (list (car prev) (cadr r)))
                  (push r merged))))
            (nreverse merged))
        comments))))
#+end_src

- *Parser-Accurate*: No regex guesswork; it uses the actual AST generated by the compiler/interpreter parser of the language.
